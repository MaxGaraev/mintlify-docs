### Path

<ParamField path="datastoreId" type="string" required>

</ParamField>

### Body

<ParamField body="id" type="string" required>
  Идентификатор источника данных, который вы хотите обновить.
</ParamField>

<ParamField body="text" type="string" required>
</ParamField>

<ParamField body="name" type="string">
  Имя источника данных.
</ParamField>

<ParamField body="metadata" type="JSON or null">
  Объект метаданных источника данных.
</ParamField>

### Response

<ResponseField name="id" type="string"></ResponseField>
